Remember we need to define the column units for two breakpoints.

Let's be semantic with Mixins


    @import "bootstrap"

    // my page module layout

    .my-module-main-body{
        .make-row();

        .sidebar{
            .make-sm-column(4);
        }

        .main-content{
            .make-sm-column(8);
        }

        .item {
            .make-row();

            .avatars{
                .make-sm-column(4);
            }

            .content{
                .make-sm-column(8);
            }
        }
    }
